📝 ANALYSIS EXAMPLE

from signal_processing.signal_processor import SignalProcessor

processor = SignalProcessor()

# Synthetic signal (educational)
signal = processor.generate_synthetic_signal("Healing Bone")
result = processor.analyze(signal)

print(f"Frequency: {result['frequency_hz']:.1f} Hz")
print(f"Q-Factor: {result['q_factor']:.2f}")
print(f"Quality: {result['signal_quality']}")
print(f"Tissue: {result['tissue_type']}")
